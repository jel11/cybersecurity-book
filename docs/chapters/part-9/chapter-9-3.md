### Python: автоматизация рутинных задач L1

```python
#!/usr/bin/env python3
"""
SOC L1 Daily Toolkit
Набор утилит для автоматизации рутинных задач
"""

import re
import ipaddress
import requests
from datetime import datetime, timedelta
from typing import Optional

# ============================================================
# 1. ПАРСЕР ЛОГОВ NGINX
# ============================================================

NGINX_PATTERN = re.compile(
    r'(?P<ip>[\d.]+)\s+-\s+-\s+\[(?P<time>[^\]]+)\]\s+'
    r'"(?P<method>\w+)\s+(?P<path>[^\s]+)\s+HTTP/[\d.]+"\s+'
    r'(?P<status>\d+)\s+(?P<size>\d+)'
)

SUSPICIOUS_PATTERNS = [
    (r'\.\./', 'Path Traversal'),
    (r"'.*OR.*'", 'SQL Injection'),
    (r'<script', 'XSS Attempt'),
    (r'cmd=|exec=|system=|passthru=', 'Command Injection'),
    (r'\.php\?.*=http', 'RFI Attempt'),
    (r'wp-login\.php|phpMyAdmin|\.env', 'Sensitive File Access'),
    (r'union.*select|select.*from', 'SQL Injection'),
    (r'base64_decode|eval\(', 'Code Execution'),
]

def analyze_nginx_log(log_file: str, threshold: int = 100) -> dict:
    """
    Анализирует nginx access.log на предмет аномалий.
    Возвращает: подозрительные IP, атаки, ошибки 4xx/5xx.
    """
    ip_stats = {}
    attacks_found = []
    
    with open(log_file, 'r', encoding='utf-8', errors='ignore') as f:
        for line in f:
            match = NGINX_PATTERN.match(line)
            if not match:
                continue
            
            ip = match.group('ip')
            path = match.group('path')
            status = int(match.group('status'))
            
            # Считаем запросы по IP
            if ip not in ip_stats:
                ip_stats[ip] = {'count': 0, 'errors': 0, 'attacks': 0}
            ip_stats[ip]['count'] += 1
            
            if status >= 400:
                ip_stats[ip]['errors'] += 1
            
            # Проверяем на атаки
            for pattern, attack_type in SUSPICIOUS_PATTERNS:
                if re.search(pattern, path, re.IGNORECASE):
                    ip_stats[ip]['attacks'] += 1
                    attacks_found.append({
                        'ip': ip,
                        'attack': attack_type,
                        'path': path[:100],
                        'status': status
                    })
    
    # Выявляем аномальные IP
    suspicious_ips = {
        ip: stats for ip, stats in ip_stats.items()
        if stats['count'] > threshold or stats['attacks'] > 0
    }
    
    return {
        'suspicious_ips': suspicious_ips,
        'attacks': attacks_found,
        'total_ips': len(ip_stats)
    }


# ============================================================
# 2. ПРОВЕРКА IP НА ПРИВАТНОСТЬ
# ============================================================

def is_private_ip(ip_str: str) -> bool:
    """Проверяет, является ли IP приватным."""
    try:
        ip = ipaddress.ip_address(ip_str)
        return ip.is_private or ip.is_loopback or ip.is_link_local
    except ValueError:
        return False


# ============================================================
# 3. ГЕНЕРАТОР ОТЧЁТА СМЕНЫ
# ============================================================

class ShiftReportGenerator:
    """Генерирует отчёт по итогам смены SOC-аналитика."""
    
    def __init__(self, analyst_name: str):
        self.analyst_name = analyst_name
        self.incidents = []
        self.false_positives = []
        self.start_time = datetime.now()
    
    def add_incident(self, title: str, severity: str, status: str, notes: str = ""):
        self.incidents.append({
            'title': title,
            'severity': severity,
            'status': status,
            'notes': notes,
            'time': datetime.now().strftime('%H:%M')
        })
    
    def add_false_positive(self, rule: str, reason: str):
        self.false_positives.append({'rule': rule, 'reason': reason})
    
    def generate(self) -> str:
        end_time = datetime.now()
        duration = end_time - self.start_time
        
        report = f"""
================================================================================
ОТЧЁТ СМЕНЫ SOC L1
================================================================================
Аналитик: {self.analyst_name}
Начало:   {self.start_time.strftime('%Y-%m-%d %H:%M')}
Конец:    {end_time.strftime('%Y-%m-%d %H:%M')}
Длительность: {str(duration).split('.')[0]}

СТАТИСТИКА:
-----------
Всего обработано алертов: {len(self.incidents) + len(self.false_positives)}
Подтверждённых инцидентов: {len(self.incidents)}
Ложных срабатываний: {len(self.false_positives)}

ИНЦИДЕНТЫ:
----------"""
        
        for i, inc in enumerate(self.incidents, 1):
            report += f"\n{i}. [{inc['severity']}] {inc['title']}"
            report += f"\n   Статус: {inc['status']}"
            report += f"\n   Время: {inc['time']}"
            if inc['notes']:
                report += f"\n   Заметки: {inc['notes']}"
            report += "\n"
        
        if self.false_positives:
            report += "\nЛОЖНЫЕ СРАБАТЫВАНИЯ:\n" + "-"*20
            for fp in self.false_positives:
                report += f"\n• Правило: {fp['rule']}"
                report += f"\n  Причина FP: {fp['reason']}\n"
        
        report += "\n" + "="*80
        return report


# Пример использования
if __name__ == "__main__":
```

## 9.3.6 Коммуникация в SOC и эскалация

### Когда эскалировать

```
ДЕРЕВО РЕШЕНИЙ: ЭСКАЛИРОВАТЬ ИЛИ НЕТ?
=========================================

Получил алерт
     |
     v
Это Critical severity?
  ДА → Эскалировать НЕМЕДЛЕННО, уведомить голосом
  НЕТ ↓
     |
Атакован критический актив (DC, DB, backup)?
  ДА → Эскалировать в течение 30 минут
  НЕТ ↓
     |
Есть признаки успешной компрометации (TP)?
  ДА → Эскалировать в течение 30 минут
  НЕТ ↓
     |
Атака продолжается и я не могу её остановить?
  ДА → Эскалировать и спросить совета
  НЕТ ↓
     |
Я не уверен — TP или FP?
  ДА → Спросить совета у старшего коллеги
  НЕТ ↓
     |
Обработать самостоятельно по playbook
```

### Как правильно эскалировать

**Формула хорошей эскалации (SBAR-модель):**

```
S — SITUATION (Ситуация):
"Я расследую алерт HIGH приоритета — RDP брутфорс на DC-01"

B — BACKGROUND (Контекст):
"847 попыток за 15 минут с IP 103.45.67.89 (известный вредоносный,
AbuseIPDB 95%). Атакуются 5 разных аккаунтов — это password spray"

A — ASSESSMENT (Оценка):
"Пока нет успешных входов, но IP всё ещё активен.
Вектор атаки — открытый RDP на DC из интернета"

R — RECOMMENDATION (Рекомендация):
"Рекомендую заблокировать IP на периметровом FW и закрыть
3389 из внешней сети. Нужна ваша авторизация для блокировки"
```

**НЕ эскалируйте так:**
> "Там какой-то алерт, посмотри" ← НЕТ

**Эскалируйте так:**
> "[HIGH] RDP брутфорс на DC-01: 847 попыток за 15 минут с 103.45.67.89 (AbuseIPDB 95%). Password spray на 5 аккаунтов. Успешных входов нет, атака продолжается. Нужна авторизация для блокировки на FW и закрытия RDP наружу."

### Общение в команде: неформальные правила

```
ПРАВИЛА КОМАНДНОЙ КОММУНИКАЦИИ:
==================================
1. ВЕДИТЕ ЗАПИСИ: всё важное — в тикет/вики, не только в голове
2. KEEP INFORMED: если уходите на перерыв во время инцидента —
   предупредите коллег, передайте контекст
3. НЕ ЗАМАЛЧИВАЙТЕ: ошибку лучше признать сразу, чем скрывать
4. SHARE KNOWLEDGE: интересный кейс? Напишите short writeup для команды
5. SHIFT HANDOVER: при передаче смены — письменный briefing об открытых тикетах
```

**Шаблон передачи смены:**

```
HANDOVER СМЕНЫ [Дата] [Время]
От: Иванов А.С. (дневная смена)
Кому: Петров В.С. (ночная смена)

ОТКРЫТЫЕ ТИКЕТЫ:
-----------------
1. ALT-2024-001337 [HIGH] RDP Brute Force DC-01
```

## 9.3.7 KPI аналитика L1: что измеряется

### Основные метрики

```
ДАШБОРД KPI SOC L1 АНАЛИТИКА:
================================

┌─────────────────────────────────────────────────────┐
│                МОИ KPI ЗА МЕСЯЦ                     │
├─────────────────────┬───────────┬───────────────────┤
│ Метрика             │ Цель      │ Мой результат     │
├─────────────────────┼───────────┼───────────────────┤
│ MTTD (время детект) │ < 15 мин  │ 12 мин ✓          │
│ MTTR (время ответа) │ < 30 мин  │ 28 мин ✓          │
│ SLA Compliance      │ > 95%     │ 97.3% ✓           │
│ Алертов обработано  │ > 150/мес │ 183 ✓             │

│ Обучение (часов)    │ > 4 ч/мес │ 6 ч ✓             │
└─────────────────────┴───────────┴───────────────────┘

MTTD = Mean Time to Detect
MTTR = Mean Time to Respond
```

### Как управлять своими KPI

**Улучшение MTTD (время до обнаружения):**
```
Факторы, влияющие на MTTD:
- Правильная приоритизация очереди (Critical/High первыми)
- Не тратить время на "очевидные" FP
- Горячие клавиши в SIEM (знайте их!)
- Шаблоны запросов под рукой
```

**Улучшение качества тикетов:**
```
Чеклист качественного тикета:
[ ] Есть ли дата/время инцидента?
[ ] Описаны ли все IoC?
[ ] Проведено ли обогащение (VT, AbuseIPDB)?
[ ] Есть ли маппинг на MITRE ATT&CK?
[ ] Описаны ли ВСЕ предпринятые действия?
[ ] Чётко указан статус и следующие шаги?
```

## 9.3.8 Месяц 1: Типичные кейсы и как их обрабатывать

### Топ-10 кейсов, которые встретите в первый месяц

```
#1: PHISHING EMAIL
===================
Симптомы: Пользователь пожаловался на подозрительное письмо
Действия:
  1. Получить образец письма (через email security gateway или от пользователя)
  2. Проверить sender domain, SPF/DKIM/DMARC
  3. Проверить ссылки: VirusTotal, URLScan.io (не открывать!)
  4. Проверить вложения (хэш): VirusTotal, AnyRun sandbox
  5. Определить масштаб: сколько получили это письмо?
  6. Если вредоносное: block domain на email gateway, уведомить пользователей

#2: MALWARE ALERT ОТ АНТИВИРУСА
================================
Симптомы: AV отчитался об обнаружении троянца на workstation
Действия:
  1. Проверить хэш файла в VirusTotal
  2. Карантин файла через EDR (или запросить у IT)
  3. Проверить Process Tree: откуда был запущен файл?
  4. Проверить сетевые соединения с этого хоста
  5. Проверить persistence (реестр, планировщик задач)
  6. Если просто детект без выполнения → может быть FP
  7. Если был запущен → расследование компрометации

#3: АНОМАЛЬНЫЙ ВХОД (НЕУРОЧНОЕ ВРЕМЯ)
======================================
Симптомы: Правило сработало на вход в 3:00 ночи
Действия:
  1. Какой пользователь? Это сервисный аккаунт или реальный?
  2. Откуда вход? IP, hostname, страна (геоблокировка?)
  3. Нормально ли это для данного пользователя? (baseline)
  4. Что делал после входа? Какие команды/процессы?
  5. Проверить VPN-логи: был ли он подключён через VPN?
  6. При сомнениях — звонить пользователю / его руководителю

#4: СКАНИРОВАНИЕ ПОРТОВ ВНУТРИ СЕТИ
=====================================
Симптомы: Один хост "стучится" на многие порты других хостов
Действия:
  1. Это может быть плановое сканирование Nessus — проверить!
  2. Кто владелец хоста-источника? Это легитимный сканер?
  3. Если нет — возможно lateral movement или compromised host
  4. Проверить процессы на хосте-источнике через EDR
  5. Какие порты сканируются? (445 SMB, 3389 RDP, 22 SSH?)
  6. При подозрении на компрометацию — эскалировать

#5: DATA EXFILTRATION ПОДОЗРЕНИЕ
==================================
Симптомы: Большой объём трафика наружу с конкретного хоста
Действия:
  1. Куда идёт трафик? IP/домен — проверить на VT
  2. Какой процесс генерирует трафик? (через EDR/netflow)
  3. Это может быть: backup в облако, CDN, телеметрия ПО
  4. Проверить DLP: что за данные передаются?
  5. Соответствует ли время рабочим часам пользователя?
  6. При подозрении — заблокировать соединение и эскалировать
```

## 9.3.11 Типичные ошибки новичков и как их избежать

```
АНТИПАТТЕРНЫ SOC-НОВИЧКА:
===========================

ОШИБКА #1: "Я разберусь сам, не хочу беспокоить коллег"
РЕШЕНИЕ: Спрашивать — это нормально. Молчать при непонимании опасно.

ОШИБКА #2: Игнорировать "тихие" алерты Medium/Low
РЕШЕНИЕ: Часто APT-атаки начинаются с Low-severity событий.
          Не закрывайте без минимальной проверки.

ОШИБКА #3: "Это очевидный False Positive" — и закрыть без проверки
РЕШЕНИЕ: Документируйте ПОЧЕМУ вы считаете это FP.
          Через неделю вы или коллеги должны понять ход мысли.

ОШИБКА #4: Открывать ссылки/вложения из расследуемых писем
РЕШЕНИЕ: НИКОГДА на рабочей машине. Только в sandbox:
          any.run, hybrid-analysis.com, app.any.run

ОШИБКА #5: Забыть обновить тикет
РЕШЕНИЕ: Правило: любое действие → запись в тикет.
          "Если не задокументировано — не существовало"

ОШИБКА #6: Паниковать при Critical алерте
РЕШЕНИЕ: Дышите. Методично: собрать контекст → эскалировать.
          Паника замедляет, не ускоряет.

ОШИБКА #7: Не следить за своими KPI
РЕШЕНИЕ: Раз в неделю проверяйте свою статистику.
          Если SLA нарушается — разберитесь почему.
```

```
ЗОЛОТЫЕ ПРАВИЛА ПЕРВЫХ 90 ДНЕЙ:
=================================

1. DOKUMENTИРУЙТЕ ВСЁ
```
